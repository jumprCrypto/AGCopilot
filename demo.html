const utils = await import('https://raw.githubusercontent.com/jumprCrypto/AGCopilot/copilot/fix-b158a735-8afc-4934-8fab-116fc94aa92a/modules/utils.js'); const { sleep, formatMcap } = utils; console.log(formatMcap(1500000));
